run{
  ALLOCATE CHANNEL MIICH1 TYPE DISK;
  ALLOCATE CHANNEL MIICH2 TYPE DISK;
  ALLOCATE CHANNEL MIICH3 TYPE DISK;
  ALLOCATE CHANNEL MIICH4 TYPE DISK;
  ALLOCATE AUXILIARY CHANNEL AUX1 TYPE DISK;
  ALLOCATE AUXILIARY CHANNEL AUX2 TYPE DISK;
  ALLOCATE AUXILIARY CHANNEL AUX3 TYPE DISK;
  ALLOCATE AUXILIARY CHANNEL AUX4 TYPE DISK;
  SET NEWNAME FOR DATABASE TO '/DATA/DRCRMDSSC/DATAFILE/%b';
  SET NEWNAME FOR tempfile 1 TO '/DATA/DRCRMDSSC/TEMPFILE/%b';
  DUPLICATE TARGET DATABASE FOR STANDBY FROM ACTIVE DATABASE DORECOVER NOFILENAMECHECK;
  SWITCH DATAFILE ALL;
  SWITCH TEMPFILE ALL;
  RELEASE CHANNEL MIICH1;
  RELEASE CHANNEL MIICH2;
  RELEASE CHANNEL MIICH3;
  RELEASE CHANNEL MIICH4;
  RELEASE CHANNEL AUX1;
  RELEASE CHANNEL AUX2;
  RELEASE CHANNEL AUX3;
  RELEASE CHANNEL AUX4;
}
